0) Boot Ubuntu Live CD
1) shrink windows partition
2) create two new partitions 
- linux (exf4, as big as possible)
- boot (fat32, boot flag)

2.1) lsblk
3) cryptsetup -v luksFormat /dev/sda2
4) sudo cryptsetup luksOpen /dev/sda2 sda2-dec
5) sudo mkfs.ext4 /dev/mapper/sda2-dec

6) install ubuntu 

encrypted decrypted partition -> /
boot parition -> /boot
esp -> /boot/efi

7) "continue testing" 
8) mount a lot of stuff, then chroot...

sudo mount /dev/mapper/sda2-dec /mnt
sudo mount /dev/sda1 /mnt/boot/efi (vfat)
sudo mount /dev/sda5 /mnt/boot (has boot flag, ext4)
cd /mnt

sudo mount --bind /sys sys
sudo mount --bind /dev dev
sudo mount --bind /run run
sudo mount --bind /proc proc
sudo chroot /mnt
